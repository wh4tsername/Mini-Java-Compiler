class Factorial {
  public static void main() {
    new Fac().Print(new Fac().ComputeFac(10));
  }
}

class Fac {
  public int ComputeFac(int num) {
    assert(num > -1);
    int num_aux;
    if (num == 0)
      num_aux = 1;
    else
      num_aux = num * this.ComputeFac(num-1);
    return num_aux;
  }

  public void Print(int arg) {
    System.out.println(arg);
  }
}
